Check Key

POST /api/checkkey
Content-Type: application/json
{
  "key": "YOUR_KEY"
}

200 {"success": true, "msg": "Key valid", "expires": "DD/MM/YYYY"}
404 {"success": false, "msg": "Key not found"}
410 {"success": false, "msg": "Key expired", "expires": "DD/MM/YYYY"}

curl

curl -s -X POST http://localhost:5000/api/checkkey \
  -H 'Content-Type: application/json' \
  -d '{"key":"YOUR_KEY"}'

python

import requests
r = requests.post('http://localhost:5000/api/checkkey', json={'key':'YOUR_KEY'})
print(r.json())

List Keys

POST /api/listkey
Content-Type: application/json
{
  "email": "you@example.com",
  "password": "yourpass"
}

curl

curl -s -X POST http://localhost:5000/api/listkey \
  -H 'Content-Type: application/json' \
  -d '{"email":"you@example.com","password":"yourpass"}'

python

import requests
r = requests.post('http://localhost:5000/api/listkey', json={'email':'you@example.com','password':'yourpass'})
print(r.json())

Edit Key

POST /api/editkey
Content-Type: application/json
{
  "email": "you@example.com",
  "password": "yourpass",
  "old_key": "OLD",
  "new_key": "NEW",
  "dd/mm/yyyy": "03/02/2000"
}

curl

curl -s -X POST http://localhost:5000/api/editkey \
  -H 'Content-Type: application/json' \
  -d '{"email":"you@example.com","password":"yourpass","old_key":"OLD","new_key":"NEW","dd/mm/yyyy":"03/02/2000"}'

python

import requests
payload = {
  'email':'you@example.com',
  'password':'yourpass',
  'old_key':'OLD',
  'new_key':'NEW',
  'dd/mm/yyyy':'03/02/2000'
}
r = requests.post('http://localhost:5000/api/editkey', json=payload)
print(r.json())
